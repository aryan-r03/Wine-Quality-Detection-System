class WineQualityModel:
    def __init__(self, config):
        """Initialize model with configuration"""
        self.scaler = StandardScaler()
        self.classifier = RandomForestClassifier(
            n_estimators=config.model.n_estimators,
            random_state=config.model.random_state
        )
        self.accuracy = 0.0
    
    def train(self, df):
        """Train the model on wine dataset"""
        # 1. Prepare features and labels
        X = df[EXPECTED_FEATURES]
        y = (df['quality'] >= 7).astype(int)  # Good if >=7
        
        # 2. Scale features
        X_scaled = self.scaler.fit_transform(X)
        
        # 3. Split data
        X_train, X_test, y_train, y_test = train_test_split(
            X_scaled, y, test_size=0.2, random_state=42
        )
        
        # 4. Train classifier
        self.classifier.fit(X_train, y_train)
        
        # 5. Evaluate
        y_pred = self.classifier.predict(X_test)
        self.accuracy = accuracy_score(y_test, y_pred)
        
        return self.accuracy
    
    def predict(self, features):
        """Predict wine quality from features"""
        # 1. Scale features
        X = self.scaler.transform([features])
        
        # 2. Get prediction
        prediction = self.classifier.predict(X)[0]
        probabilities = self.classifier.predict_proba(X)[0]
        
        # 3. Format result
        return {
            'quality': 'Good Wine' if prediction else 'Bad Wine',
            'confidence': max(probabilities) * 100,
            'good_probability': probabilities[1] * 100,
            'bad_probability': probabilities[0] * 100
        }
    
    def save(self, filepath):
        """Save model to file"""
        with open(filepath, 'wb') as f:
            pickle.dump(self, f)
    
    @staticmethod
    def load(filepath):
        """Load model from file"""
        with open(filepath, 'rb') as f:
            return pickle.load(f)
```

class DataHandler:

class ModelManager:
    def __init__(self, config):
        self.config = config
        self.model_path = 'wine_model.pkl'
    
    def get_model(self):
        """Get model (load existing or train new)"""
        # Try to load existing model
        model = self._load_existing_model()
        
        if model and self._validate_model(model):
            print("âœ“ Model loaded successfully")
            return model
        
        # Train new model if load failed or invalid
        print("Training new model...")
        return self._train_new_model()
    
    def _load_existing_model(self):
        """Load model from disk"""
        try:
            return WineQualityModel.load(self.model_path)
        except:
            return None
    
    def _validate_model(self, model):
        """Validate model compatibility"""
        # Check if model has required attributes
        if not hasattr(model, 'scaler'):
            return False
        if not hasattr(model, 'classifier'):
            return False
        
        # Check feature compatibility
        try:
            test_features = [0] * len(EXPECTED_FEATURES)
            model.scaler.transform([test_features])
            return True
        except:
            return False
    
    def _train_new_model(self):
        """Train and save new model"""
        # Load or generate data
        data_handler = DataHandler()
        df = data_handler.load_csv('winequality.csv')
        
        if df is None:
            df = data_handler.generate_synthetic_data()
        
        # Train model
        model = WineQualityModel(self.config)
        accuracy = model.train(df)
        print(f"Model trained! Accuracy: {accuracy:.2%}")
        
        # Save model
        model.save(self.model_path)
        
        return model
```
